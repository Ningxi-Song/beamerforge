\section{Overview}

\begin{frame}{The deck should start with one clear claim}
\begin{itemize}
  \setlength{\itemsep}{6pt}
  \item State the question in one line.
  \item Name the setting or application.
  \item Preview the main evidence.
  \item End with the implication.
\end{itemize}
\vfill
\begin{block}{Replacement rule}
Keep the structure, then swap in your own argument and examples.
\end{block}
\end{frame}

\begin{frame}{A reusable talk needs a simple arc}
\begin{enumerate}
  \setlength{\itemsep}{7pt}
  \item Motivation
  \item Research design
  \item Main result
  \item Interpretation
\end{enumerate}
\vfill
\begin{exampleblock}{Template note}
Use short titles, sparse bullets, and one message per slide.
\end{exampleblock}
\end{frame}
