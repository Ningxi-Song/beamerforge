\section{Overview}

\begin{frame}{Motivation}
  \begin{itemize}
    \item State the research question or presentation objective in one sentence.
    \item Summarize the setting, data, or problem that motivates the talk.
    \item Preview the main result or contribution.
  \end{itemize}
\end{frame}

\begin{frame}{Roadmap}
  \begin{enumerate}
    \item Background and research question
    \item Data, method, or design
    \item Main evidence
    \item Interpretation and next steps
  \end{enumerate}
\end{frame}

\begin{frame}{Template Building Blocks}
  This slide demonstrates all three block types available in the D Rose theme.

  \begin{block}{Main point}
    Use a regular block for the key claim or finding on a slide.
  \end{block}

  \begin{exampleblock}{Example}
    Use an example block for empirical illustrations, definitions, or intuition.
  \end{exampleblock}

  \begin{alertblock}{Caution}
    Use an alert block for assumptions, limitations, or unresolved questions.
  \end{alertblock}
\end{frame}

\begin{frame}{Bullet Hierarchy}
  \begin{itemize}
    \item First-level item uses the arrow-head bullet
    \begin{itemize}
      \item Second-level item uses a smaller arrow
      \begin{itemize}
        \item Third-level item (default bullet)
      \end{itemize}
    \end{itemize}
    \item Another top-level point
  \end{itemize}

  \vspace{0.5cm}
  Enumerate style:
  \begin{enumerate}
    \item Numbered item one
    \item Numbered item two
    \begin{enumerate}
      \item Lettered sub-item
      \item Another sub-item
    \end{enumerate}
  \end{enumerate}
\end{frame}
