\documentclass[10pt]{beamer}

% Standard bullet list content pattern

% Colors
\definecolor{primary}{RGB}{186,84,68}
\setbeamercolor*{structure}{fg=primary}
\setbeamercolor{frametitle}{fg=primary,bg=white}

% Custom bullets
\usepackage{pifont}
\setbeamertemplate{itemize item}{\ding{220}}
\setbeamertemplate{itemize subitem}{\ding{216}}

% Document
\title[]{Bullet List Demonstration}
\author[]{Author Name}
\institute[]{Institution Name}
\date{}

\begin{document}

\begin{frame}
  \titlepage
\end{frame}

\begin{frame}{Standard Bullet List}
  \begin{itemize}
    \item First point (one line)
    \item Second point (one line)
    \item Third point (one line)
    \item Fourth point (one line)
    \item Fifth point (one line)
  \end{itemize}
\end{frame}

\begin{frame}{With Sub-bullets}
  \begin{itemize}
    \item Main point one
      \begin{itemize}
        \item Sub-point A
        \item Sub-point B
      \end{itemize}
    \item Main point two
      \begin{itemize}
        \item Sub-point C
        \item Sub-point D
      \end{itemize}
    \item Main point three
  \end{itemize}
\end{frame}

\begin{frame}{Rules for Bullet Lists}
  \begin{itemize}
    \item 3-5 bullets maximum per slide
    \item One line per bullet (no wrapping)
    \item One phrase per bullet (no paragraphs)
    \item Consistent grammar throughout
  \end{itemize}
\end{frame}

\begin{frame}{Conclusion}
  \begin{block}{Key Takeaway}
    Bullet lists are the most common content pattern. Keep them short and consistent.
  \end{block}
\end{frame}

\end{document}
