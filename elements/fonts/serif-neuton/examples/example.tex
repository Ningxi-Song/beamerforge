\documentclass[10pt]{beamer}

% Theme: typography/serif-neuton
% Neuton serif font family

% Language
\RequirePackage[english]{babel}
\RequirePackage{fontspec}
\usefonttheme{professionalfonts}
\usefonttheme{serif}

% Font setup
\IfFileExists{../fonts/Neuton-Regular.ttf}{%
  \setmainfont[
    Path = ../fonts/,
    UprightFont = Neuton-Regular.ttf,
    ItalicFont = Neuton-Italic.ttf,
    BoldFont = Neuton-Regular.ttf
  ]{Neuton}%
  \setsansfont[
    Path = ../fonts/,
    UprightFont = Neuton-Regular.ttf,
    ItalicFont = Neuton-Italic.ttf,
    BoldFont = Neuton-ExtraBold.ttf
  ]{Neuton}%
}{%
  \setmainfont{Latin Modern Roman}%
  \setsansfont{Latin Modern Sans}%
}

% Colors (minimal for font demo)
\definecolor{primary}{RGB}{186,84,68}
\setbeamercolor*{structure}{fg=primary}
\setbeamercolor{frametitle}{fg=primary,bg=white}
\setbeamercolor{title}{fg=primary,bg=white}

% Title styling
\setbeamerfont{frametitle}{shape=\scshape}
\setbeamerfont{title}{size=\fontsize{20pt}{32pt}\selectfont}
\setbeamerfont{author}{size=\fontsize{12pt}{32pt}\selectfont}
\setbeamerfont{institute}{size=\fontsize{12pt}{32pt}\selectfont}

% Document
\title[]{Typography Demonstration}
\author[]{Author Name}
\institute[]{Institution Name}
\date{}

\begin{document}

\begin{frame}
  \titlepage
\end{frame}

\begin{frame}{Neuton Serif Font Family}
  \begin{itemize}
    \item Regular weight for body text
    \item Italic for emphasis
    \item ExtraBold for titles and strong emphasis
    \item Small caps via \texttt{\textbackslash scshape} for frame titles
  \end{itemize}
\end{frame}

\begin{frame}{Font Weights}
  \begin{block}{Regular Weight}
    This is regular Neuton text. It provides excellent readability on screen.
  \end{block}
  
  \begin{exampleblock}{Light Weight}
    \fontspec[Path=../fonts/,Extension=.ttf,UprightFont=*-Light]{Neuton}This is light Neuton text. Use for captions and footnotes.
  \end{exampleblock}
  
  \begin{alertblock}{ExtraBold Weight}
    \fontspec[Path=../fonts/,Extension=.ttf,UprightFont=*-ExtraBold]{Neuton}This is extrabold Neuton text. Use for titles and strong emphasis.
  \end{alertblock}
\end{frame}

\begin{frame}{Typographic Hierarchy}
  \begin{enumerate}
    \item Title: 20pt, small caps
    \item Frame title: small caps, primary color
    \item Body: regular weight, 11pt
    \item Caption: light weight, footnotesize
  \end{enumerate}
\end{frame}

\begin{frame}{Conclusion}
  \begin{block}{Key Takeaway}
    Neuton provides elegant, professional typography for academic presentations.
  \end{block}
\end{frame}

\end{document}
