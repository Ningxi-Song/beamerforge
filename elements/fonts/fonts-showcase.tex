% Helper to show a font sample or fallback note
\newcommand{\fontslide}[3]{%
  \IfFontExistsTF{#1}{%
    \fontspec{#1}%
    \begin{frame}{#2: #1}
      \textbf{Family:} #1 \hfill \textit{#3}

      \vspace{0.3cm}
      {\Large The quick brown fox jumps over the lazy dog.}

      \vspace{0.2cm}
      \textbf{The quick brown fox jumps over the lazy dog.}

      \vspace{0.2cm}
      \textit{The quick brown fox jumps over the lazy dog.}
    \end{frame}
  }{}%
}
