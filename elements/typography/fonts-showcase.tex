\section{How to Choose}
% ============================================================

\begin{frame}{How to Choose a Font}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Serif fonts} feel:
    \begin{itemize}
      \item Academic, traditional
      \item Authoritative, formal
      \item Best for: humanities, history, law
    \end{itemize}

    \vspace{0.5cm}
    \textbf{For your template:}
    \begin{enumerate}
      \item Pick one serif family
      \item Pick one sans family
      \item Decide which is primary
      \item Bundled fonts = no dependency
    \end{enumerate}

    \column{0.5\textwidth}
    \textbf{Sans-serif fonts} feel:
    \begin{itemize}
      \item Modern, clean
      \item Technical, data-driven
      \item Best for: economics, policy, STEM
    \end{itemize}

    \vspace{0.5cm}
    \textbf{Quick tip:}
    \begin{itemize}
      \item Pair a serif heading with sans body (or vice versa)
      \item Test at 10pt -- the real slide size
      \item Avoid script/display fonts for body text
    \end{itemize}
  \end{columns}
\end{frame}
